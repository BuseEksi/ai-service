name: Testler

# main'e push'ta ve main'e açılan her pull request'te çalışır
on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest

    steps:
      - name: Repoyu checkout et
        uses: actions/checkout@v4

      - name: Python 3.12 kur
        uses: actions/setup-python@v5
        with:
          python-version: "3.12"
          cache: "pip"

      - name: Bağımlılıkları yükle
        run: pip install -r requirements-dev.txt

      - name: Testleri çalıştır
        run: pytest -v
